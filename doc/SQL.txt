
/* ユーザー情報 */
DROP TABLE User IF EXISTS;
CREATE TABLE User (
 id  varchar(12) PRIMARY KEY NOT NULL,
 name varchar(20) NOT NULL,
 password varchar (20) NOT NULL
);
/* アイテムマスター */
DROP TABLE Item IF EXISTS;
CREATE TABLE Item (
 id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 user_id varchar(20) NOT NULL,
 parts_code int NOT NULL,
 pattern int NOT NULL,
 rain int NOT NULL,
 wind int NOT NULL,
 photo varchar(100) NOT NULL
);
/* 日記 */
DROP TABLE Diary IF EXISTS;
CREATE TABLE Diary (
 id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 date date NOT NULL,
 user_id varchar(12) NOT NULL,
 note varchar(140),
 photo varchar(100) NOT NULL,
 weather_code int NOT NULL,
 max_temperature double NOT NULL,
 min_temperature double NOT NULL,
 wind_speed double NOT NULL,
 amount_of_rain double NOT NULL
);
/* おすすめコーデ */
DROP TABLE Recommends IF EXISTS;
CREATE TABLE Recommends (
 id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 user_id varchar(12) NOT NULL,
 date date NOT NULL,
 outer int,
 jacket int,
 tops  int,
 skirt  int,
 pants int,
 shoes int
);
/* 季節マスター */
DROP TABLE Season IF EXISTS;
CREATE TABLE Season (
 id int PRIMARY KEY NOT NULL,
 code int NOT NULL,
 name varchar(10) NOT NULL,
 upperlimit_temperature int NOT NULL,
 lowerlimit_temperature int NOT NULL,
 start_month int NOT NULL,
 end_month int NOT NULL
);
/* アイテム_季節設定マスター */
DROP TABLE Item_season IF EXISTS;
CREATE TABLE Item_season (
 id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 item_id int NOT NULL,
 code int NOT NULL,
 flag int NOT NULL
);
/* 色マスター */
DROP TABLE Color IF EXISTS;
CREATE TABLE Color (
 code int PRIMARY KEY NOT NULL,
 name varchar(10) NOT NULL
);
/* アイテム_色設定マスター */
DROP TABLE Item_color IF EXISTS;
CREATE TABLE Item_color (
 id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
 item_id int NOT NULL,
 code int NOT NULL,
 flag int NOT NULL
);
/* パーツマスター */
DROP TABLE Parts IF EXISTS;
CREATE TABLE Parts (
 code int PRIMARY KEY NOT NULL,
 name varchar(10) NOT NULL
);
/* 天気マスター */
DROP TABLE Weather IF EXISTS;
CREATE TABLE Weather (
 code int PRIMARY KEY NOT NULL,
 name varchar(10) NOT NULL
);
